/* 부서 */
CREATE TABLE TABLE (
	DEPTNO NUMBER(2) NOT NULL, /* 부서번호 */
	DNAME VARCHAR2(14), /* 부서이름 */
	LOC VARCHAR2(13) /* 지역 */
);

COMMENT ON TABLE TABLE IS '부서';

COMMENT ON COLUMN TABLE.DEPTNO IS '부서번호';

COMMENT ON COLUMN TABLE.DNAME IS '부서이름';

COMMENT ON COLUMN TABLE.LOC IS '지역';

CREATE UNIQUE INDEX PK_TABLE
	ON TABLE (
		DEPTNO ASC
	);

ALTER TABLE TABLE
	ADD
		CONSTRAINT PK_TABLE
		PRIMARY KEY (
			DEPTNO
		);
